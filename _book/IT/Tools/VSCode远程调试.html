Redis远程调试-配置文件

  • tasks.json

    {
    "version": "2.0.0",
    "tasks": [
        {
            "type": "shell",
            "label": "gcc",
            "command": "make",

            "options": {
                "cwd": "${workspaceFolder}"
            },
            "problemMatcher": [
                "$gcc"
            ],
            "group": "build"
        }
    ]
}
